API
- Get Topic
	
	Description: Get topic data

	Method: GET
	Endpoint: /getTopic
	Cookie: session-id
	respond: [
  		{
    		"TopicName": "<Topic1Name>",
    		"FieldColor": "<Topic1FieldColor>",
    		"posX": "<Topic1PosX>",
    		"posY": "<Topic1PoxY>",
		"DocCount": "<Topic1DocumentCount>"
  		},...
  		{
    		"TopicName": "<Topic#nName>",
    		"FieldColor": "<Topic#nFieldColor>",
    		"posX": "<Topic#nPosX>",
    		"posY": "<Topic#nPoxY>",
		"DocCount": "<Topic#nDocumentCount>"
  		}
		]


- Get all Document in Topic
	
	Description: Get document name from topic name
	Endpoint: /getDocID/<topic>&<topic2>...
	Cookie: session-id
	respond: [
		{"TopicName": "<Topic#1Name>", "DocID": ["<docID#1>","<docID#n>"]},
		{"TopicName": "<Topic#nName>", "DocID": ["<docID#1>","<docID#n>"]}
		]


- Get Document data

	Description: Fetch document data from document's name

	Method: GET
	Endpoint: /getDocData/<docID>&<docID2>
	Cookie: session-id
	respond: {
		"Document name": "<DocName>"
		"Content": "<content>",
		"Download Count": "<Download count>",
		}


- Download file

	Description: Download document from document's id

	Method: GET
	Endpoint: /getDoc/<docID>
	Cookie: session-id
	respond: <Document>

- Upload file

	Description: Upload Document

	Method: POST
	Endpoint: /postDoc/

	Payload : <metadata.json>,<payload.txt>
	Cookie: session-id
	
	respond: "Uploaded"

- Delete file

	Description: Delete Document

	Method: GET
	Endpoint: /delDoc/<docID>
	Cookie: session-id
	
	respond: "Document Deleted"

- Add Topic

	Description: Add topic

	Method: GET
	Endpoint: /addTopic
	Cookie: 
	Payload: {"TopicName":"<TopicName>",
		  "FieldColor":"<TopicFieldColor>",
		  "PosX":"<TopicPositionX>",
		  "PosX":"<TopicPositionY>",
		 }

	respond: "Topic added"
